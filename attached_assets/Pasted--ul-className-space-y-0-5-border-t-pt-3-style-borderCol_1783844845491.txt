                }}

                  activeNav === "settings" ? "text-white" : cx(c.textMuted, c.surfaceHover)
                )}
                style={activeNav === "settings" ? { backgroundColor: "#FF6B35" } : undefined}
              >
                <Settings size={17} strokeWidth={2} />

          </ul>
        </nav>

        <div className={cx("border-t px-5 py-4", c.border)}>

          </div>
          <p className={cx("mt-1 text-[11px]", c.textFaint)}>Version 2.4.0</p>

      </aside>
    </>
  );
}
